    { text: "─────────────────────", action: null },
    { text: "🔢 Matrice Num 3 niveaux", action: () => this.convertToMatrixNum3() },
    { text: "🔢 Matrice Num 4 niveaux", action: () => this.convertToMatrixNum4() },
