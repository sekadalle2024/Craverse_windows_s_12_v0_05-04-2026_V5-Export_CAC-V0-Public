ÉTAPE 3: GESTION DES SÉPARATEURS DANS createMenuItem()
═══════════════════════════════════════════════════════════════════════════════

Localiser la méthode createMenuItem() dans menu.js (ligne ~200) et MODIFIER pour gérer
les séparateurs visuels:

createMenuItem(item) {
  // Gérer les séparateurs
  if (!item.action && item.text.includes('─')) {
    const separator = document.createElement("div");
    separator.className = "accordion-separator";
    separator.style.cssText = `
      height: 1px;
      background: linear-gradient(to right, transparent, rgba(56, 1, 1, 0.2), transparent);
      margin: 8px 18px;
    `;
    return separator;
  }

  const menuItem = document.createElement("div");
  menuItem.className = "accordion-item";
  menuItem.style.cssText = `display: flex; justify-content: space-between; align-items: center; padding: 10px 18px 10px 44px; cursor: pointer; transition: all 0.2s ease; color: #666; font-size: 13px;`;
  menuItem.innerHTML = `<span>${item.text}</span>${item.shortcut ? `<span style="font-size: 11px; color: #888; background: rgba(56, 1, 1, 0.1); padding: 2px 6px; border-radius: 4px;">${item.shortcut}</span>` : ''}`;

  menuItem.addEventListener("mouseenter", () => { menuItem.style.background = "rgba(56, 1, 1, 0.15)"; menuItem.style.color = "#1A0000"; });
  menuItem.addEventListener("mouseleave", () => { menuItem.style.background = "transparent"; menuItem.style.color = "#666"; });
  menuItem.addEventListener("click", (e) => {
    e.stopPropagation();
    if (item.action) { try { item.action(); } catch (err) { this.showAlert(`❌ Erreur: ${err.message}`); } }
    this.hideMenu();
  });
  return menuItem;
}

ÉTAPE 4: MISE À JOUR DE LA MÉTHODE applyRiskStyle()
═══════════════════════════════════════════════════════════════════════════════

La méthode applyRiskStyle() existante doit être compatible avec tous les formats.
Vérifier qu'elle utilise bien normalizeRiskValue() qui détecte automatiquement le format.

Si nécessaire, mettre à jour pour supporter les nouveaux formats:

applyRiskStyle(cell, riskLevel) {
  if (!cell) return;

  // Essayer d'abord les couleurs Alpha 3
  let colors = this.getColorsAlpha3();
  let normalizedLevel = this.normalizeToAlpha3(riskLevel);
  
  // Si pas trouvé, essayer Alpha 5
  if (!normalizedLevel) {
    colors = this.getColorsAlpha5();
    normalizedLevel = this.normalizeToAlpha5(riskLevel);
  }
  
  // Si pas trouvé, essayer Alpha 4
  if (!normalizedLevel) {
    colors = this.getColorsAlpha4();
    normalizedLevel = this.normalizeToAlpha4(riskLevel);
  }
  
  // Si pas trouvé, essayer numérique
  if (!normalizedLevel) {
    const numValue = parseFloat(riskLevel);
    if (!isNaN(numValue)) {
      if (numValue <= 9) colors = this.getColorsNumeric(9);
      else if (numValue <= 16) colors = this.getColorsNumeric(16);
      else if (numValue <= 25) colors = this.getColorsNumeric(25);
      normalizedLevel = numValue;
    }
  }

  if (normalizedLevel && colors[normalizedLevel]) {
    cell.style.setProperty('background-color', colors[normalizedLevel].bg, 'important');
    cell.style.setProperty('color', colors[normalizedLevel].text, 'important');
    cell.style.setProperty('font-weight', '600', 'important');
    cell.style.setProperty('text-align', 'center', 'important');
    cell.style.setProperty('padding', '8px 12px', 'important');
  }
}
